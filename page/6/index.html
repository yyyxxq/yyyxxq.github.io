NOI2014 购票

2018-07-05
#NOI #斜率优化 #CDQ分治 #点分治

题目大意：

    有一棵树，对于每个点v有四个参数,svs_vs​v​​,pvp_vp​v​​,qvq_vq​v​​,lvl_vl​v​​。

    svs_vs​v​​表示当前点到父亲节点的距离。如果d为每次跳跃的距离，那么每次跳跃的费用为dpv+qvdp_v+q_vdp​v​​+q​v​​。现在所有点都要去到1号点，并且他们每次往上跳不能超过距离lvl_vl​v​​。对于每个点求最小花费。

    n≤2∗105n\leq2*10^5n≤2∗10​5​​,其余所有运算不超过longlong。

题解：

    链上做法：斜率优化

    设did_id​i​​为1号点到i号点的距离

    fi=min(fj+(di−dj)pi+qi)f_i=min(f_j+(d_i-d_j)p_i+q_i)f​i​​=min(f​j​​+(d​i​​−d​j​​)p​i​​+q​i​​)

    设选k由于j且dk>djd_k>d_jd​k​​>d​j​​：

    fk+(di−dk)pi+qi<fj+(di−dj)pi+qif_k+(d_i-d_k)p_i+q_i<f_j+(d_i-d_j)p_i+q_if​k​​+(d​i​​−d​k​​)p​i​​+q​i​​<f​j​​+(d​i​​−d​j​​)p​i​​+q​i​​

    整理得：

    fk−fjdk−dj<pi\frac{f_k-f_j}{d_k-d_j}<p_i​d​k​​−d​j​​​​f​k​​−f​j​​​​<p​i​​

    树链剖分+凸包合并O(nlog3n)O(n\log^3 n)O(nlog​3​​n)：

    对于一条链来说，直接斜率优化即可，那么对于树上，我们可以按照dfs的顺序，一次加入点，并将树树剖，依次求出f值，在插入线段树的时候，只有左右儿子都是满时才合并，因为没加入的都不会被查询，所以可在(nlog3n)(n\log^3n)(nlog​3​​n)的时间里求出答案（树剖一条链会有log段，限制lil_il​i​​在线段树上取区间总共有log个，每个区间二分得到答案又一个log）（因为树剖常数小所以能过qwq）（然而懒惰的我写了下面的方法。）

    CDQ+树分治+凸包合并O(nlog2n)O(n\log^2n)O(nlog​2​​n)：

    对于每次分治，把重心堵住，不停地堵住，然后利用当前节点到重心的链去更新重心答案，再递归计算。（看代码应该极好理解）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 420000
#define slope(x,y) ((f[y]-f[x]+0.0)/(d[y]-d[x]))
#define cal(i,j) (0LL+f[j]+(d[i]-d[j])*p[i]+q[i])
using namespace std;
long long n,fa[N],tot,h[N],v[N],Next[N],siz[N],g[N],cnt,Q[N],o[N];
long long p[N],q[N],L[N],d[N],w[N],f[N];
struct node{long long w,id;}a[N];
bool cmp(node x,node y){return x.w>y.w;}
int add(int x,int y,long long z){
    Next[++tot]=h[x];
    h[x]=tot;
    v[tot]=y;
    w[tot]=z;
}
int dfs(int x){
    siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        d[v[i]]=d[x]+w[i];
        dfs(v[i]);
        siz[x]+=siz[v[i]];
    }
}
int get(int x,int S,int &rt){ 
    g[x]=0; siz[x]=1;
    for(int i=h[x];i;i=Next[i]){
        if(o[v[i]])continue;
        get(v[i],S,rt);
        siz[x]+=siz[v[i]];
        g[x]=max(g[x],siz[v[i]]);
    }
    g[x]=max(g[x],S-siz[x]);
    if(g[x]<g[rt]&&siz[x]>1)rt=x;
}
int dfs2(int x){
    a[++cnt].id=x;
    a[cnt].w=d[x]-L[x];
    for(int i=h[x];i;i=Next[i])if(!o[v[i]])dfs2(v[i]);
}
int solve(int x,int S){
    if(x==2){
        x++;
        x--;
    }
    if(S==1)return 0;
    int rt=0,now;
    get(x,S,rt);
    for(int i=h[rt];i;i=Next[i])o[v[i]]=1;
    solve(x,S-siz[rt]+1);
    cnt=0;
    for(int i=h[rt];i;i=Next[i])dfs2(v[i]);
    sort(a+1,a+cnt+1,cmp);
    now=rt;
    int top,l,r,mid,pos;
    top=0;
    for(int i=1;i<=cnt;i++){
        while(now!=fa[x]&&d[a[i].id]-L[a[i].id]<=d[now]){
            while(top>1&&slope(Q[top],now)>=slope(Q[top-1],Q[top]))top--;
            Q[++top]=now;now=fa[now];
        }
        if(top>0){
            l=1;r=top;pos=1;
            while(l<=r){
                mid=(l+r)>>1;
                if(mid==top){pos=top;break;}
                if(slope(Q[mid],Q[mid+1])>=p[a[i].id])l=mid+1,pos=mid+1;
                else r=mid-1;
            }
            f[a[i].id]=min(f[a[i].id],cal(a[i].id,Q[pos]));
        }
    }
    for(int i=h[rt];i;i=Next[i])solve(v[i],siz[v[i]]);
}
int main(){
    long long x;
    scanf("%lld%lld",&n,&x);
    for(int i=2;i<=n;i++){
        scanf("%lld%lld%lld%lld%lld",&fa[i],&x,&p[i],&q[i],&L[i]);
        add(fa[i],i,x);
    }
    dfs(1);
    g[0]=n+1;
    for(int i=2;i<=n;i++)f[i]=1LL<<60;
    solve(1,siz[1]);
    for(int i=2;i<=n;i++)printf("%lld\n",f[i]);
}

NOI2014 随机数生成器

2018-07-02
#NOI #卡常 #贪心 #模拟

题目大意：

    给出一个随机生成的矩阵（数字为1到n*m），求字典序最小的排序后的从(1,1)(1,1)(1,1)到(n,m)(n,m)(n,m)的数字序列。

题解：

    按照题意模拟出来之后从小到大贪心，每次加入点之后把不可能的情况排除，直接暴力排除（遇到被排除的就退），因为每个点最多只会被排除一次，所以排除的复杂度是O(n∗m)O(n*m)O(n∗m)的，模拟出矩阵也是O(n∗m)O(n*m)O(n∗m)的，总复杂度为O(n∗m)O(n*m)O(n∗m)。

#include<iostream>
#include<stdio.h>
#include<string.h>
#define swap(x,y) t=x;x=y;y=t;
using namespace std;
int x0,a,b,c,d,n,m,q,tot;
int f[26000000],p[26000000],l[10200],r[10200];
static int t;
int main(){
    scanf("%d%d%d%d%d",&x0,&a,&b,&c,&d);
    scanf("%d%d%d",&n,&m,&q);
    for(int i=1;i<=n*m;i++)f[i]=i;
    for(int i=1;i<=n*m;i++){
        x0=(1LL*a*x0*x0+1LL*b*x0+c)%d;
        swap(f[i],f[x0%i+1]);
    }
    static int x,y;
    for(static int i=1;i<=q;i++){
        scanf("%d%d",&x,&y);
        swap(f[x],f[y]);
    }
    for(static int i=1;i<=n*m;i++)p[f[i]]=i;
    for(static int i=1;i<=n*m;i++){
        bool ooo=true;
        int y=(p[i]-1)%m+1;
        int x=(p[i]-y)/m+1;
        if(p[i]==0)continue;
        for(int i=x-1;i>=1;i--){
            for(int j=y+1;j<=m;j++){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        for(int i=x+1;i<=n;i++){
            for(int j=y-1;j>=1;j--){
                int ooo=((i-1)*m+j);
                if(p[f[ooo]]==0)break;
                p[f[ooo]]=0;
            }
        }
        if(ooo==false)continue;
        l[++tot]=x;
        r[tot]=y;
        printf("%d ",i);
        if(tot==n+m-1)return 0; 
    }
}

NOI2014 魔法森林

2018-07-02
#NOI #SPFA

题目大意：

    一个无向图，每条边两个值a和b，求1到n路径上的min(max(ai)+max(bj))min(max(a_i)+max(b_j))min(max(a​i​​)+max(b​j​​))。

题解：

    按照a为第一关键字，b为第二关键字排序，按顺序加边，再松弛，因为每条边松弛一次，所以跟整张图的SPFA是一个复杂度。相当于对于每个a的可能值都做一次b的最短路，约等于优化暴力（spaly维护最小生成树的做法我不会啊qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 12000000
using namespace std;
struct node{
    int x,y,a,b;
}a[N];
int tot,Next[N],v[N],w[N],h[N],d[N]; 
int q[N],p[N],n,m;
int add(int x,int y,int z){
    tot++;
    Next[tot]=h[x];
    v[tot]=y;
    w[tot]=z;
    h[x]=tot;
}
bool cmp(node x,node y){
    if(x.a==y.a)return x.b<y.b;
    return x.a<y.a;
}
int spfa(int x){
    q[1]=x;
    if(x==1){
        x++;
        x--;
    }
    int head=0,tail=1;
    p[x]=1;
    while(head<tail){
        int now=q[++head];
        if(d[now]==0)continue;
        for(int i=h[now];i;i=Next[i]){
            if(max(d[now],w[i])<d[v[i]]||d[v[i]]==0){
                d[v[i]]=max(d[now],w[i]);
                if(p[v[i]]==0){
                    p[v[i]]=1;
                    q[++tail]=v[i];
                }
            }
        }
        p[now]=0;
    }
    p[x]=0;
}
int main(){
    d[1]=1;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)scanf("%d%d%d%d",&a[i].x,&a[i].y,&a[i].a,&a[i].b);
    sort(a+1,a+1+m,cmp);
    int ans=1000000000,bns=1000000000;
    for(int i=1;i<=m;i++){
        add(a[i].x,a[i].y,a[i].b);
        add(a[i].y,a[i].x,a[i].b);
        spfa(a[i].x);
        spfa(a[i].y);
        if(d[n]==0)continue;
        bns=a[i].a+d[n];
        if(bns<ans)ans=bns;
    }
    spfa(1);
    if(ans==1000000000)ans=-1;
    printf("%d",ans);
}

#倍增 #NOI #KMP #卡常

题目大意：

    对于S的每个前缀T，对于T求出有多少个长度小于|T|的前缀和后缀相同，讲每个前缀T的答案加一乘起来膜100000000710000000071000000007。

题解：

    直接求Next数组然后倍增就好了。（倍增前后坐标相反能快差不多一倍，不然卡不过去qwq）

#include<iostream>
#include<stdio.h>
#include<string.h>
#define N 1200000
#define P 1000000007
using namespace std;
char s[N];
long long f[21][N],ans;
int Next[N],q[N],g[N];
int cal(int x){
    if(f[0][x]==0)return 0;
    int o=x;
    for(int i=20;i>=0;i--)
        if(f[i][o]>x/2)o=f[i][o];
    return f[0][o];
}
int get(){
    memset(f,0,sizeof(f));
    memset(q,0,sizeof(q));
    memset(g,0,sizeof(g));
    memset(Next,0,sizeof(Next));
    int l=strlen(s+1);
    q[1]=1;
    for(int i=2;i<=l;i++){
        int j=Next[i-1];
        while(j&&s[j+1]!=s[i])j=Next[j];
        Next[i]=j+(s[j+1]==s[i]);
        q[i]=q[Next[i]]+1;
        f[0][i]=Next[i];
        for(int j=1;j<=20;j++){
            f[j][i]=f[j-1][f[j-1][i]];
            if(f[j][i]==0)break;
        }
    }
    ans=1;
    for(int i=1;i<=l;i++)g[i]=q[cal(i)];
    for(int i=1;i<=l;i++)(ans*=1LL*g[i]+1)%=P; 
    printf("%lld\n",ans);
}
int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){

NOI2014 起床困难综合征

#NOI #贪心

题目大意：

    给定一系列位运算操作，在1到m里选一个数，使这个数在经历一系列操作后值最大。

题解：

    对于每一位分别求出输入时是0和1为多少，从高位往低位贪心选择较大值，如果在相同满足要求的结果下选择输入较小的。

#include<iostream>
#include<stdio.h>
#include<string.h> 
using namespace std;
long long n,m,d,ans,l;
char s[100];
int main(){
    scanf("%lld%lld",&n,&m);
    long long x=0,y=(1LL<<32)-1;
    for(int i=1;i<=n;i++){
        scanf("%s",s+1);
        scanf("%lld",&d);
        if(s[1]=='O')x=x|d,y=y|d;
        if(s[1]=='A')x=x&d,y=y&d;
        if(s[1]=='X')x=x^d,y=y^d;
    }
    for(int i=32;i>=0;i--){
        if(x&(1LL<<(i))){
            ans+=1LL<<i;
        }else{
            if(y&(1LL<<i)){
                if(l+(1LL<<i)<=m){
                    ans+=1LL<<i;
                    l+=1LL<<i;
                }
            }
        }
    }
    printf("%lld",ans);
}

NOI2007 生成树计数

2018-06-30
#NOI #状压DP #最小表示法

题目大意：

    有一条链，链上长度不超过k的点之间都会进行连边，求生成树个数。

题解：

    60%做法：O(n3)O(n^3)O(n​3​​),n≤100n \leq 100n≤100

    那么直接连边按照，用矩阵树定理高斯消元解行列式即可。

 因为n≤100n\leq100n≤100那么直接连边按照，用矩阵树定理n3n^3n​3​​高斯消元解行列式即可。

    100%做法：O(l3log(n))O(l^3log(n))O(l​3​​log(n)),n≤1015n\leq 10^{15}n≤10​15​​,l为所有联通形式种类，当k=5时l=52。

    考虑到对于第i-k号点，如果最后会成为生成树，他必定与i-k+1到i号点中至少一个是一个联通快的。因此，我们维护fi,jf_{i,j}f​i,j​​表示，在当前第i个点，前j个点的联通块是怎么样的，每次转移则是用第i号点与前面的连边情况进行转移，维护判断i-k号点在转移后是否与i-k+1到i号点中的部分点联通。搜索我们可以得到转移矩阵。

    初始条件：因为在前k个点中，所有点之间都是两两相连的，n个点的完全图的最小生成树是nn−2n^{n-2}n​n−2​​，这个可用purfer序列简易证明。

    关于最小表示法：既联通情况2,2,3,1与联通情况1,1,2,3是相同的，所有本质相同的情况我们用它的最小表示来识别它。

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<map>
#define N 120
#define P 65521
using namespace std;
int cnt,k,tot,l;
int a[10],b[10],q[60];
long long n;
map<int,int>p;
struct matrix{
    long long a[N][N];
    int clear(){memset(a,0,sizeof(a));}
    int E(){for(int i=1;i<=cnt;i++)a[i][i]=1;}
}g,f;
matrix operator *(matrix a,matrix b){
    matrix ans;
    ans.clear();
    for(int i=1;i<=cnt;i++){
        for(int j=1;j<=cnt;j++){
            for(int k=1;k<=cnt;k++){
                (ans.a[i][j]+=a.a[i][k]*b.a[k][j])%=P;
            }
        }
    }
    return ans;
}
matrix power(matrix x,long long k){
    matrix ans;
    ans.clear();
    ans.E();
    while(k){
        if(k&1)ans=ans*x;
        x=x*x;
        k>>=1;
    }
    return ans;
}
int get(int x){
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    int tot=0;
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3);
    for(int i=1;i<=k;i++){
        if(b[a[i]])a[i]=b[a[i]];
        else b[a[i]]=++tot,a[i]=tot;
    }
    int y=0;
    for(int i=1;i<=k;i++)y=y<<3,y+=a[i];
    return y;
}
int powerr(int x,int k){
    int ans=1;
    if(k<=0)return ans;
    while(k){
        if(k&1)(ans*=x)%=P;
        (x*=x)%=P;
        k>>=1;
    }
    return ans;
}
int gett(int x){
    int ans=1;
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3),b[a[i]]++;
    for(int i=1;i<=k;i++)ans*=powerr(b[i],b[i]-2);
    return ans;
}
int dfs(int x,int o){
    if(x==l){
        int y=get(o);
        if(p[y]==0){
            p[y]=++cnt;
            q[cnt]=y;
            f.a[1][cnt]=gett(y);
        }
        return 0;
    }
    for(int i=1;i<=k;i++){
        dfs(x+1,(o<<3)+i);
    }
}
int check(int x,int y){
    int o=0;
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=1;i<=k;i++)a[i]=(x>>(3*(k-i)))%(1<<3);
    for(int i=2;i<=k;i++){
        if(a[i]==a[1])break;
        if(y&1)continue;
        if(i==k)return 0;
    }
    for(int i=0;i<k;i++){
        if((1<<i)&y){
            if(o==0)o=a[i+1];
            if(b[a[i+1]]>0)return 0;
            b[a[i+1]]=o;
        }
    }
    for(int i=1;i<=k;i++)if(b[a[i]]>0)a[i]=o;
    int l=0;
    for(int i=2;i<=k;i++)l=(l<<3)+a[i];
    l<<=3;
    if(y!=0)l+=o;
    else l+=k+1;
    int ans=get(l);
    return p[ans];
}
int cal(int x){
    for(int i=1;i<=cnt;i++){
        int l=check(q[i],x);
        if(l>0){
            g.a[i][l]++;
        }
    }
}
int main(){
    scanf("%d%lld",&k,&n);
    if(k>n){
        printf("%d",powerr(n,n-2));
        return 0;
    }
    l=k;
    dfs(0,0);
    for(int i=0;i<(1<<k);i++)cal(i);
    f=f*power(g,n-k);
    printf("%d\n",f.a[1][1]);
}

NOI2007 项链工厂

2018-06-28
#NOI #线段树

题目大意：

   维护一个环，和一个指针，有以下几种操作（该指针的下标为1，其余为位置的下标顺时针递增）:

   1:将指针逆时针移动k位

   2:将环以指针做对称轴翻转

   3:交换l位置和r位置上的颜色

   4:将l到r染成c颜色

   5:查询当前项链有多少段颜色（首尾相同算一段）

   6: 查询l到r有多少段颜色

题解：

题目大意：

    股市有n天，有A和B两只股票，给出aia_ia​i​​，bib_ib​i​​表示A劵和B劵的价值，rir_ir​i​​表示每天提供的A劵与B劵的比例（必须按照这个比例来买）。

题解：

    易证每天必须全部卖出或者全部买入

    60%做法，O(n2)dpO(n^2)dpO(n​2​​)dp：

    fi=max(fi−1,Aj∗ai+Bj∗bi)f_i=\max(f_{i-1},A_j*a_i+B_j*b_i)f​i​​=max(f​i−1​​,A​j​​∗a​i​​+B​j​​∗b​i​​)

    100%做法，O(nlog2n)O(nlog^2n)O(nlog​2​​n)斜率优化+cdq分治:

    若k优于j且AkA_kA​k​​>AjA_jA​j​​，则：

    Aj∗ai+Bj∗bi<Ak∗ai+Bk∗biA_j*a_i+B_j*b_i<A_k*a_i+B_k*b_iA​j​​∗a​i​​+B​j​​∗b​i​​<A​k​​∗a​i​​+B​k​​∗b​i​​

    bi∗(Bj−Bk)<ai∗(Ak−Aj)b_i*(B_j-B_k)<a_i*(A_k-A_j)b​i​​∗(B​j​​−B​k​​)<a​i​​∗(A​k​​−A​j​​)

    −Ak−AjBk−Bj<aibi-\frac{A_k-A_j}{B_k-B_j}<\frac{a_i}{b_i}−​B​k​​−B​j​​​​A​k​​−A​j​​​​<​b​i​​​​a​i​​​​

    Ak−AjBk−Bj>−aibi\frac{A_k-A_j}{B_k-B_j}>-\frac{a_i}{b_i}​B​k​​−B​j​​​​A​k​​−A​j​​​​>−​b​i​​​​a​i​​​​

    CDQ分治：solve(l,r)

    先按斜率−aibi-\frac{a_i}{b_i}−​b​i​​​​a​i​​​​排序。在solve(l,r)把时间小于等于mid的放到数组前半段，大于mid的放到数组后半段，solve(l,mid)，并将其暴力构建凸包，对mid+1到r的答案进行更新（这时满足斜率单调的条件，而最先开始的按照斜率排序保证了后半段的查询中斜率式单调递增的）。然后solve(mid+1,r)，最后将l到r中的点按照横纵坐标排序（为了返回之后能构建凸包）。

    ans.x=max(ans.x,s1.query(1,1,n,st[y],st[x],d));
    ans.y=max(ans.y,s2.query(1,1,n,st[y],st[x],d));

    return ans;
}

    illyasviel a,b;

    b=query_illyasviel(y,z,d);
    return max(a.x,b.x)+max(a.y,b.y);
}
double query(int x,int y){
    double l=0.0,r=2000000.0;
//    printf("%.5lf\n",check((double)4.85,x,y));
    while(r-l>1e-5){
        double mid=(l+r)/2;
        if(check(mid,x,y)>eps)l=mid;
        else r=mid;
    }
    return l+1e-6;
}
