Mushroom 追妹纸

2018-07-31
#hash #二分

题目描述：

    Mushroom最近看上了一个漂亮妹纸。他选择一种非常经典的手段来表达自己的心意——写情书。考虑到自己的表达能力，Mushroom决定不手写情书。他从网上找到了两篇极佳的情书，打算选择其中共同的部分。另外，Mushroom还有个一个情敌Ertanis，此人也写了封情书给妹子。

    Mushroom不希望自己的情书中完整的出现了情敌的情书。（这样抄袭的事情就暴露了）。

    Mushroom把两封情书分别用字符串s1和s2来表示，Ertanis的情书用字符串s3来表示，他要截取的部分用字符串w表示。

需满足：

    1、w是s1的子串

    2、w是s2的子串

    3、s3不是w的子串

    4、w的长度应尽可能大

    所谓子串是指：在字符串中连续的一段

题解：

    发现答案具有二分性，又发现可以O（n）扫描当前答案是否可行，那就直接二分答案加hash就搞完了。

#include<iostream>
#include<stdio.h>
#include<string.h>
#include<map>
#define G 37
#define P 1000000007
#define N 120000
using namespace std;
char s1[N],s2[N],w[N];
long long n,m,k,a[N],b[N],pw[N],oo,h;
map<int,int> o;
int pre(){
    h=0;
    pw[0]=1;
    for(int i=1;i<=max(n,m);i++)pw[i]=(1LL*pw[i-1]*G)%P;
    for(int i=1;i<=k;i++)h=(h*G+w[i]-'a'+1)%P;
    oo=0;
    for(int i=1;i<=k;i++)oo=(oo*G%P+s1[i]-'a'+1)%P;
    a[1]=oo==h;
    for(int i=k+1;i<=n;i++){
        oo=((oo-pw[k-1]*(s1[i-k]-'a'+1)%P))*G+s1[i]-'a'+1;
        oo=oo%P;
        if(oo<0)oo+=P;
        a[i-k+1]=oo==h;
    }
}
int check(int x){
    if(x==0)return 1;
    o.clear();
    h=0;
    int u=0;
    int f;
    for(int i=1;i<=x;i++){
        if(i>=k)u+=a[i-k+1];
        h=(1LL*h*G+s1[i]-'a'+1)%P;
    }
    if(u==0||x<k)o[h]=1;
//  printf("%d ",h);
    for(int i=x+1;i<=n;i++){
        if(i>=k)u+=a[i-k+1];
        u-=a[i-x];
        h=(h-pw[x-1]*(s1[i-x]-'a'+1)%P)*G+s1[i]-'a'+1;
        h=h%P;
        if(h<0)h+=P;
//  printf("%d ",h);
        if(u==0||x<k)o[h]=1;
    }
//  printf("\n");
    oo=0;
    for(int i=1;i<=x;i++)oo=(1LL*oo*G+s2[i]-'a'+1)%P;
    if(o[oo]==1)return 1;
//  printf("%d ",oo);
    for(int i=x+1;i<=m;i++){
        oo=(oo-pw[x-1]*(s2[i-x]-'a'+1)%P)*G+s2[i]-'a'+1;
        oo=oo%P;
//  printf("%d ",oo);
        if(oo<0)oo+=P;
        if(o[oo]==1)return 1;
    }
//  printf("\n");
    return 0;
}
int main(){
    scanf("%s",s1+1);
    scanf("%s",s2+1);
    scanf("%s",w+1);
    n=strlen(s1+1),m=strlen(s2+1),k=strlen(w+1);
    pre();
    int l=0,r=min(n,m);
    while(r-l>2){
        int mid=(l+r)/2;
        if(check(mid))l=mid;
        else r=mid-1;
    }
    for(int i=r;i>=l;i--){
        if(check(i)){
            printf("%d\n",i);
            return 0;
        }
    }
}
